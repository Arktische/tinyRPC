set(CMAKE_CXX_STANDARD 20)

        ${PROJECT_SOURCE_DIR}
)

        common_test
        common_test.cc
        trait.hpp
)

        common_test
        gtest_main
        ${GTEST_LIBRARIES}
)

        common_test
        PRIVATE -fconcepts
)

        common_bench
        common_bench.cc
)

        common_bench
        PRIVATE -O3
)

        common_bench
        benchmark::benchmark
)

        common_test
